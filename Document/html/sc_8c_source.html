130 void SC_ResetReader(SC_T *sc)
131 {
132  uint32_t u32Intf;
133 
134  if(sc == SC0)
135  u32Intf = 0;
136  else
137  return ;
138 
139  // Reset FIFO, enable auto de-activation while card removal
140  sc->ALTCTL |= (SC_ALTCTL_TXRST_Msk | SC_ALTCTL_RXRST_Msk | SC_ALTCTL_ADACEN_Msk);
141  // Set Rx trigger level to 1 character, longest card detect debounce period, disable error retry (EMV ATR does not use error retry)
142  while(sc->CTL & SC_CTL_SYNC_Msk);
143  sc->CTL &= ~(SC_CTL_RXTRGLV_Msk | SC_CTL_CDDBSEL_Msk | SC_CTL_TXRTY_Msk | SC_CTL_RXRTY_Msk);
144  // Enable auto convention, and all three smartcard internal timers
145  sc->CTL |= SC_CTL_AUTOCEN_Msk | SC_CTL_TMRSEL_Msk;
146  // Disable Rx timeout
147  sc->RXTOUT = 0;
148  // 372 clocks per ETU by default
149  sc->ETUCTL = 371;
150 
151  /* Enable necessary interrupt for smartcard operation */
152  if(u32CardStateIgnore[u32Intf]) // Do not enable card detect interrupt if card present state ignore
153  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
154  SC_INTEN_TERRIEN_Msk |
155  SC_INTEN_TMR0IEN_Msk |
156  SC_INTEN_TMR1IEN_Msk |
157  SC_INTEN_TMR2IEN_Msk |
158  SC_INTEN_BGTIEN_Msk |
159  SC_INTEN_ACERRIEN_Msk);
160  else
161  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
162  SC_INTEN_TERRIEN_Msk |
163  SC_INTEN_TMR0IEN_Msk |
164  SC_INTEN_TMR1IEN_Msk |
165  SC_INTEN_TMR2IEN_Msk |
166  SC_INTEN_BGTIEN_Msk |
167  SC_INTEN_ACERRIEN_Msk |
168  SC_INTEN_CDIEN_Msk);
169 
170  return;
171 }
