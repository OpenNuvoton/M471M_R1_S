546 void I2C_SMBusTimeout(I2C_T *i2c, uint32_t ms, uint32_t u32Pclk)
547 {

549 
550  i2c->BUSCTL &= ~I2C_BUSCTL_TIDLE_Msk;
551 
552  /* DIV4 disabled */
