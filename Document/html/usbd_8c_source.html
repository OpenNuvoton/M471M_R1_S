324  // Return current configuration setting

335  }
336  case GET_DESCRIPTOR:
337  {
338  USBD_GetDescriptor();

341  }
342  case GET_INTERFACE:
343  {
344  // Return current interface setting
345  /* Data stage */
346  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = g_usbd_UsbAltInterface;
347  USBD_SET_DATA1(EP0);
348  USBD_SET_PAYLOAD_LEN(EP0, 1);
349  /* Status stage */
350  USBD_PrepareCtrlOut(0, 0);
351 
352  DBG_PRINTF("Get interface\n");
353 
354  break;
355  }
356  case GET_STATUS:
357  {
358  // Device
359  if(g_usbd_SetupPacket[0] == 0x80)
360  {
361  uint8_t u8Tmp;
362 
363  u8Tmp = 0;
364  if(g_usbd_sInfo->gu8ConfigDesc[7] & 0x40) u8Tmp |= 1; // Self-Powered/Bus-Powered.
365  if(g_usbd_sInfo->gu8ConfigDesc[7] & 0x20) u8Tmp |= (g_usbd_RemoteWakeupEn << 1); // Remote wake up
366 
367  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = u8Tmp;
368 
369  }
370  // Interface
371  else if(g_usbd_SetupPacket[0] == 0x81)
372  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = 0;
373  // Endpoint
374  else if(g_usbd_SetupPacket[0] == 0x82)
375  {
376  uint8_t ep = g_usbd_SetupPacket[4] & 0xF;
377  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0)) = USBD_GetStall(ep) ? 1 : 0;
378  }
379 
380  M8(USBD_BUF_BASE + USBD_GET_EP_BUF_ADDR(EP0) + 1) = 0;
381  /* Data stage */
382  USBD_SET_DATA1(EP0);
383  USBD_SET_PAYLOAD_LEN(EP0, 2);
384  /* Status stage */
385  USBD_PrepareCtrlOut(0, 0);
386 
387  DBG_PRINTF("Get status\n");
388 
389  break;
390  }
391  default:
392  {
393  /* Setup error, stall the device */
394  USBD_SET_EP_STALL(EP0);
395  USBD_SET_EP_STALL(EP1);
396 
397  DBG_PRINTF("Unknown request. stall ctrl pipe.\n");
398 
399  break;
400  }
401  }
402  }
403  else
404  {
405  // Host to device
406  switch(g_usbd_SetupPacket[1])
407  {
408  case CLEAR_FEATURE:
409  {
410  if(g_usbd_SetupPacket[2] == FEATURE_ENDPOINT_HALT)
411  {
412  int32_t epNum, i;
413 
414  /* EP number stall is not allow to be clear in MSC class "Error Recovery Test".

437  }
