339 void RTC_SetDateAndTime(S_RTC_TIME_DATA_T *sPt)
340 {
341  uint32_t u32RegCAL, u32RegTIME;
342 
343  if(sPt == 0)
344  return ;
345 
346  /*-----------------------------------------------------------------------------------------------------*/
347  /* Set RTC 24/12 hour setting and Day of the Week */
348  /*-----------------------------------------------------------------------------------------------------*/
349  RTC_WaitAccessEnable();
350  if(sPt->u32TimeScale == RTC_CLOCK_12)
351  {
352  RTC->CLKFMT &= ~RTC_CLKFMT_24HEN_Msk;
353 
354  /*-------------------------------------------------------------------------------------------------*/
355  /* Important, range of 12-hour PM mode is 21 up to 32 */
356  /*-------------------------------------------------------------------------------------------------*/
357  if(sPt->u32AmPm == RTC_PM)
358  sPt->u32Hour += 20;
359  }
360  else
361  {
362  RTC->CLKFMT |= RTC_CLKFMT_24HEN_Msk;
363  }
364 
365  /* Set Day of the Week */
366  RTC->WEEKDAY = sPt->u32DayOfWeek;
367 
368  /*-----------------------------------------------------------------------------------------------------*/
369  /* Set RTC Current Date and Time */
370  /*-----------------------------------------------------------------------------------------------------*/
371  u32RegCAL = ((sPt->u32Year - RTC_YEAR2000) / 10) << 20;
372  u32RegCAL |= (((sPt->u32Year - RTC_YEAR2000) % 10) << 16);
373  u32RegCAL |= ((sPt->u32Month / 10) << 12);
374  u32RegCAL |= ((sPt->u32Month % 10) << 8);
375  u32RegCAL |= ((sPt->u32Day / 10) << 4);
376  u32RegCAL |= (sPt->u32Day % 10);
377 
378  u32RegTIME = ((sPt->u32Hour / 10) << 20);
379  u32RegTIME |= ((sPt->u32Hour % 10) << 16);
380  u32RegTIME |= ((sPt->u32Minute / 10) << 12);
381  u32RegTIME |= ((sPt->u32Minute % 10) << 8);
382  u32RegTIME |= ((sPt->u32Second / 10) << 4);
383  u32RegTIME |= (sPt->u32Second % 10);
384 
385  /*-----------------------------------------------------------------------------------------------------*/
386  /* Set RTC Calender and Time Loading */
387  /*-----------------------------------------------------------------------------------------------------*/
388  RTC_WaitAccessEnable();
389  RTC->CAL = (uint32_t)u32RegCAL;
390  RTC->TIME = (uint32_t)u32RegTIME;
